from neo4j import AsyncGraphDatabase, AsyncDriver
from config import get_settings
import logging

logger = logging.getLogger(__name__)

_driver: AsyncDriver | None = None


async def get_driver() -> AsyncDriver:
    global _driver
    settings = get_settings()
    if _driver is None:
        _driver = AsyncGraphDatabase.driver(
            settings.NEO4J_URI,
            auth=(settings.NEO4J_USER, settings.NEO4J_PASSWORD),
        )
        logger.info("Neo4j driver initialized")
    return _driver


async def close_driver():
    global _driver
    if _driver:
        await _driver.close()
        _driver = None


async def run_query(cypher: str, parameters: dict = None) -> list[dict]:
    driver = await get_driver()
    async with driver.session() as session:
        result = await session.run(cypher, parameters or {})
        records = await result.data()
        return records


async def run_write_query(cypher: str, parameters: dict = None):
    driver = await get_driver()
    async with driver.session() as session:
        result = await session.run(cypher, parameters or {})
        summary = await result.consume()
        return summary

import os
import pandas as pd
import glob
import logging
from config import get_settings

logger = logging.getLogger(__name__)


def load_csv_folder(folder_name: str) -> pd.DataFrame:
    """Load all CSV/parquet files from a folder into a single DataFrame."""
    settings = get_settings()
    folder_path = os.path.join(settings.DATA_DIR, folder_name)

    if not os.path.exists(folder_path):
        logger.warning(f"Folder not found: {folder_path}")
        return pd.DataFrame()

    files = glob.glob(os.path.join(folder_path, "**/*.csv"), recursive=True)
    files += glob.glob(os.path.join(folder_path, "**/*.parquet"), recursive=True)
    files += glob.glob(os.path.join(folder_path, "*.csv"))
    files += glob.glob(os.path.join(folder_path, "*.parquet"))

    # Deduplicate
    files = list(set(files))

    if not files:
        logger.warning(f"No CSV/parquet files in: {folder_path}")
        return pd.DataFrame()

    dfs = []
    for f in files:
        try:
            if f.endswith(".parquet"):
                df = pd.read_parquet(f)
            else:
                df = pd.read_csv(f, low_memory=False)
            dfs.append(df)
            logger.info(f"Loaded {f}: {len(df)} rows")
        except Exception as e:
            logger.error(f"Error loading {f}: {e}")

    if not dfs:
        return pd.DataFrame()

    return pd.concat(dfs, ignore_index=True)


def clean_row(row: dict) -> dict:
    """Remove NaN values from row dict."""
    return {k: (None if pd.isna(v) else v) for k, v in row.items()}


def df_to_records(df: pd.DataFrame) -> list[dict]:
    """Convert DataFrame to list of clean dicts."""
    if df.empty:
        return []
    return [clean_row(row) for row in df.to_dict(orient="records")]

BATCH_SIZE = 500


async def initialize_schema():
    """Create indexes."""
    for idx_query in SCHEMA_INDEXES:
        try:
            await run_write_query(idx_query)
        except Exception as e:
            logger.warning(f"Index creation warning: {e}")
    logger.info("Schema indexes created")


async def _batch_upsert(label: str, id_field: str, records: list[dict]):
    """Generic batch upsert for a node."""
    if not records:
        return
    for i in range(0, len(records), BATCH_SIZE):
        batch = records[i : i + BATCH_SIZE]
        cypher = f"""
        UNWIND $batch AS row
        MERGE (n:{label} {{{id_field}: row.{id_field}}})
        SET n += row
        """
        await run_write_query(cypher, {"batch": batch})
    logger.info(f"Upserted {len(records)} {label} nodes")


async def _batch_relate(
    label_a: str,
    id_a: str,
    rel: str,
    label_b: str,
    id_b: str,
    records: list[dict],
    prop_a: str = None,
    prop_b: str = None,
):
    """Create relationships in batch."""
    if not records:
        return
    prop_a = prop_a or id_a
    prop_b = prop_b or id_b
    for i in range(0, len(records), BATCH_SIZE):
        batch = records[i : i + BATCH_SIZE]
        cypher = f"""
        UNWIND $batch AS row
        MATCH (a:{label_a} {{{id_a}: row.{prop_a}}})
        MATCH (b:{label_b} {{{id_b}: row.{prop_b}}})
        MERGE (a)-[:{rel}]->(b)
        """
        await run_write_query(cypher, {"batch": batch})

import re
import logging

logger = logging.getLogger(__name__)

# Topics that are allowed
ALLOWED_TOPICS = [
    "order", "sales", "billing", "invoice", "payment", "delivery",
    "shipment", "product", "material", "customer", "partner", "plant",
    "journal", "accounting", "revenue", "o2c", "sap", "erp",
    "document", "schedule", "cancellation", "storage", "warehouse",
    "quantity", "amount", "currency", "company", "organization",
    "analyze", "show", "find", "list", "count", "total", "sum",
    "which", "what", "how", "when", "where", "who",
    "graph", "node", "relationship", "connected", "linked",
    "cash", "receivable", "ledger", "entry",
]

# Blocked patterns
BLOCKED_PATTERNS = [
    r"\b(password|secret|api.?key|token|credential)\b",
    r"\b(hack|exploit|injection|drop\s+table|delete\s+all)\b",
    r"\b(personal\s+info|social\s+security|credit\s+card)\b",
    r"(ignore\s+previous|forget\s+instructions|jailbreak|dan\s+mode)",
    r"\b(weather|recipe|joke|movie|sport|news|politics|religion)\b",
    r"\b(write\s+code|generate\s+script|create\s+program)\b(?!.*cypher)",
]

CYPHER_DANGEROUS_PATTERNS = [
    r"\bDELETE\b",
    r"\bDETACH\s+DELETE\b",
    r"\bDROP\b",
    r"\bCREATE\s+USER\b",
    r"\bCREATE\s+DATABASE\b",
    r"\bSET\s+PASSWORD\b",
    r"\bCALL\s+dbms\b",
    r"\bCALL\s+apoc\.periodic\.commit\b",
]


def check_message_safety(message: str) -> tuple[bool, str]:
    """
    Returns (is_safe, reason_if_not_safe)
    """
    msg_lower = message.lower()

    # Check blocked patterns
    for pattern in BLOCKED_PATTERNS:
        if re.search(pattern, msg_lower, re.IGNORECASE):
            return False, "I can only help with SAP Order-to-Cash data analysis. Please ask about orders, billing, deliveries, or other O2C topics."

    # Check if message has any O2C relevance (for non-trivial messages)
    words = msg_lower.split()
    if len(words) > 5:
        has_relevant = any(
            any(topic in word for topic in ALLOWED_TOPICS)
            for word in words
        )
        if not has_relevant:
            return False, "I'm specialized in SAP Order-to-Cash analysis. Please ask me about sales orders, billing documents, deliveries, payments, or related O2C topics."

    return True, ""


def validate_cypher_query(cypher: str) -> tuple[bool, str]:
    """
    Validate that a generated Cypher query is safe to execute.
    Returns (is_safe, reason_if_not_safe)
    """
    for pattern in CYPHER_DANGEROUS_PATTERNS:
        if re.search(pattern, cypher, re.IGNORECASE):
            return False, f"Generated query contains unsafe operation: {pattern}"

    # Must be a read query or safe write
    if not re.search(r"\b(MATCH|RETURN|WITH|CALL|UNWIND)\b", cypher, re.IGNORECASE):
        return False, "Query does not appear to be a valid read query"

    return True, ""


def sanitize_cypher_limit(cypher: str, max_limit: int = 500) -> str:
    """Ensure query doesn't return too many records."""
    if not re.search(r"\bLIMIT\s+\d+", cypher, re.IGNORECASE):
        # Add limit before any ORDER BY at the end, or just at the end
        cypher = cypher.rstrip().rstrip(";")
        cypher += f" LIMIT {max_limit}"
    return cypher

import json
import logging
import re
from openai import AsyncOpenAI
from config import get_settings
from database.neo4j_client import run_query
from database.schema import SCHEMA_DESCRIPTION
from llm.prompts import SYSTEM_PROMPT, CYPHER_GENERATION_PROMPT, ANSWER_GENERATION_PROMPT
from llm.guardrails import check_message_safety, validate_cypher_query, sanitize_cypher_limit
from tenacity import retry, stop_after_attempt, wait_exponential

logger = logging.getLogger(__name__)


class GraphAgent:
    def __init__(self):
        settings = get_settings()
        self.client = AsyncOpenAI(api_key=settings.OPENAI_API_KEY)
        self.model = settings.OPENAI_MODEL
        self.conversation_history = []

    @retry(stop=stop_after_attempt(3), wait=wait_exponential(min=1, max=10))
    async def _call_llm(self, messages: list[dict], response_format: str = "text") -> str:
        kwargs = {
            "model": self.model,
            "messages": messages,
            "temperature": 0.1,
        }
        if response_format == "json":
            kwargs["response_format"] = {"type": "json_object"}

        response = await self.client.chat.completions.create(**kwargs)
        return response.choices[0].message.content

    async def _generate_cypher(self, question: str) -> dict:
        """Generate a Cypher query from a natural language question."""
        prompt = CYPHER_GENERATION_PROMPT.format(
            question=question,
            schema=SCHEMA_DESCRIPTION
        )
        messages = [
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": prompt}
        ]
        raw = await self._call_llm(messages, response_format="json")
        try:
            return json.loads(raw)
        except json.JSONDecodeError:
            # Try to extract JSON from response
            match = re.search(r'\{.*\}', raw, re.DOTALL)
            if match:
                return json.loads(match.group())
            return {"cypher": None, "explanation": "Failed to generate query"}

    async def _execute_cypher(self, cypher: str) -> list[dict]:
        """Safely execute a Cypher query."""
        is_safe, reason = validate_cypher_query(cypher)
        if not is_safe:
            logger.warning(f"Unsafe Cypher blocked: {reason}")
            return []

        cypher = sanitize_cypher_limit(cypher)

        try:
            results = await run_query(cypher)
            return results
        except Exception as e:
            logger.error(f"Cypher execution error: {e}")
            raise

    async def _generate_answer(
        self,
        question: str,
        cypher: str,
        results: list[dict]
    ) -> str:
        """Generate a natural language answer from query results."""
        # Truncate results for prompt
        results_preview = results[:20] if len(results) > 20 else results
        results_str = json.dumps(results_preview, indent=2, default=str)

        prompt = ANSWER_GENERATION_PROMPT.format(
            question=question,
            cypher=cypher,
            count=len(results),
            results=results_str
        )

        messages = [
            {"role": "system", "content": SYSTEM_PROMPT},
            *self.conversation_history[-6:],  # Keep last 3 exchanges
            {"role": "user", "content": prompt}
        ]

        return await self._call_llm(messages)

    async def _extract_highlight_ids(self, results: list[dict]) -> list[dict]:
        """Extract node IDs from results for graph highlighting."""
        highlights = []
        for record in results[:50]:
            for key, val in record.items():
                if isinstance(val, dict):
                    # Neo4j node
                    labels = val.get("labels", [])
                    props = val.get("properties", {})
                    if labels and props:
                        highlights.append({
                            "label": labels[0],
                            "properties": props
                        })
        return highlights

    async def chat(self, message: str, session_id: str = "default") -> dict:
        """
        Main chat entry point.
        Returns: {answer, cypher, results, highlights, error}
        """
        # Guardrail check
        is_safe, reason = check_message_safety(message)
        if not is_safe:
            return {
                "answer": reason,
                "cypher": None,
                "results": [],
                "highlights": [],
                "error": None
            }

        try:
            # Generate Cypher
            cypher_result = await self._generate_cypher(message)
            cypher = cypher_result.get("cypher")
            explanation = cypher_result.get("explanation", "")

            results = []
            answer = ""

            if cypher:
                # Execute query
                try:
                    results = await self._execute_cypher(cypher)
                except Exception as e:
                    # Try to fix query
                    logger.error(f"Query failed, attempting fix: {e}")
                    fix_messages = [
                        {"role": "system", "content": SYSTEM_PROMPT},
                        {"role": "user", "content": f"This Cypher query failed with error '{e}'. Fix it:\n{cypher}\nReturn only the fixed query as JSON: {{\"cypher\": \"...\"}}"}
                    ]
                    fix_raw = await self._call_llm(fix_messages, response_format="json")
                    fix_data = json.loads(fix_raw)
                    fixed_cypher = fix_data.get("cypher", cypher)
                    results = await self._execute_cypher(fixed_cypher)
                    cypher = fixed_cypher

                # Generate answer
                answer = await self._generate_answer(message, cypher, results)
            else:
                # Direct conversational response
                messages = [
                    {"role": "system", "content": SYSTEM_PROMPT},
                    *self.conversation_history[-6:],
                    {"role": "user", "content": message}
                ]
                answer = await self._call_llm(messages)

            # Update conversation history
            self.conversation_history.append({"role": "user", "content": message})
            self.conversation_history.append({"role": "assistant", "content": answer})

            # Extract highlights
            highlights = await self._extract_highlight_ids(results)

            return {
                "answer": answer,
                "cypher": cypher,
                "results": results[:100],
                "highlights": highlights,
                "error": None
            }

        except Exception as e:
            logger.error(f"Agent error: {e}", exc_info=True)
            return {
                "answer": "I encountered an error processing your request. Please try again.",
                "cypher": None,
                "results": [],
                "highlights": [],
                "error": str(e)
            }

):
    """Get graph overview for visualization."""
    if node_types:
        labels = [l.strip() for l in node_types.split(",")]
        label_filter = "|".join(labels)
        node_query = f"""
        MATCH (n:{label_filter})

        """

            'BusinessPartner', 'JournalEntry', 'Payment', 'Product'
        ])}
        WITH n LIMIT {limit}
        RETURN 
            id(n) AS id,
            labels(n) AS labels,
            properties(n) AS props
        """

    nodes_raw = await run_query(node_query)

    # Get edges between these nodes
    node_ids = [r["id"] for r in nodes_raw]
    if node_ids:
        edge_query = """
        MATCH (a)-[r]->(b)
        WHERE id(a) IN $ids AND id(b) IN $ids
        RETURN 
            id(a) AS source,
            id(b) AS target,
            type(r) AS rel_type
        LIMIT 2000
        """
        edges_raw = await run_query(edge_query, {"ids": node_ids})
    else:
        edges_raw = []

    # Format nodes
    nodes = []
    for r in nodes_raw:
        props = r.get("props", {})
        label = r.get("labels", ["Unknown"])[0]
        # Get display name
        display_id = (
            props.get("SalesOrder") or
            props.get("BillingDocument") or
            props.get("OutboundDelivery") or
            props.get("BusinessPartner") or
            props.get("AccountingDocument") or
            props.get("Product") or
            props.get("Plant") or
            str(r["id"])
        )
        nodes.append({
            "id": str(r["id"]),
            "label": label,
            "display_id": str(display_id),
            "properties": {k: str(v) if v is not None else None for k, v in props.items()},
        })

    edges = [
        {
            "source": str(e["source"]),
            "target": str(e["target"]),
            "type": e["rel_type"]
        }
        for e in edges_raw
    ]

    return GraphData(
        nodes=nodes,
        edges=edges,
        node_count=len(nodes),
        edge_count=len(edges)
    )

    global _ingestion_status
    _ingestion_status["running"] = True
    _ingestion_status["completed"] = False
    _ingestion_status["error"] = None
    _ingestion_status["results"] = []
    try:
        results = await run_full_ingestion()
        _ingestion_status["results"] = results
        _ingestion_status["completed"] = True
    except Exception as e:
        logger.error(f"Ingestion failed: {e}", exc_info=True)
        _ingestion_status["error"] = str(e)
    finally:
        _ingestion_status["running"] = False


@router.post("/start", response_model=IngestionStatus)

    """Trigger full data ingestion pipeline."""
    if _ingestion_status["running"]:
        raise HTTPException(status_code=409, detail="Ingestion already running")
    background_tasks.add_task(_run_ingestion_task)

        status = "completed"

@app.get("/api/folders")
async def list_folders():
    """List available data folders."""
    data_dir = settings.DATA_DIR
    if not os.path.exists(data_dir):
        return {"folders": []}
    folders = [
        f for f in os.listdir(data_dir)
        if os.path.isdir(os.path.join(data_dir, f))
    ]
    result = []
    for folder in sorted(folders):
        folder_path = os.path.join(data_dir, folder)
        files = []
        for root, _, fs in os.walk(folder_path):
            for file in fs:
                fp = os.path.join(root, file)
                size = os.path.getsize(fp)
                mtime = os.path.getmtime(fp)
                files.append({"name": file, "size": size, "modified": mtime})
        result.append({"name": folder, "file_count": len(files), "files": files})
    return {"folders": result, "total": len(result)}

import { useState, useCallback, useRef } from "react";
import { fetchGraphOverview, fetchNodeDetail, fetchGraphStats } from "../services/api";

export function useGraph() {
  const [graphData, setGraphData] = useState({ nodes: [], edges: [] });
  const [selectedNode, setSelectedNode] = useState(null);
  const [highlightedIds, setHighlightedIds] = useState(new Set());
  const [loading, setLoading] = useState(false);
  const [stats, setStats] = useState(null);
  const [error, setError] = useState(null);
  const abortRef = useRef(null);

  const loadGraph = useCallback(async (limit = 300, nodeTypes = null) => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetchGraphOverview(limit, nodeTypes);
      setGraphData(res.data);
    } catch (e) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }, []);

  const loadNodeDetail = useCallback(async (nodeId) => {
    try {
      const res = await fetchNodeDetail(nodeId);
      setSelectedNode({ id: nodeId, ...res.data });
    } catch (e) {
      console.error("Node detail error:", e);
    }
  }, []);

  const loadStats = useCallback(async () => {
    try {
      const res = await fetchGraphStats();
      setStats(res.data);
    } catch (e) {
      console.error("Stats error:", e);
    }
  }, []);

  const highlightNodes = useCallback((highlights) => {
    if (!highlights || highlights.length === 0) {
      setHighlightedIds(new Set());
      return;
    }
    // highlights is [{label, properties}]
    const ids = new Set();
    graphData.nodes.forEach((node) => {
      highlights.forEach((h) => {
        if (node.label === h.label) {
          const props = h.properties || {};
          const nodeProps = node.properties || {};
          const matches = Object.entries(props).some(
            ([k, v]) => nodeProps[k] && String(nodeProps[k]) === String(v)
          );
          if (matches) ids.add(node.id);
        }
      });
    });
    setHighlightedIds(ids);
  }, [graphData.nodes]);

  const clearHighlights = useCallback(() => setHighlightedIds(new Set()), []);

  return {
    graphData,
    selectedNode,
    highlightedIds,
    loading,
    stats,
    error,
    loadGraph,
    loadNodeDetail,
    loadStats,
    highlightNodes,
    clearHighlights,
    setSelectedNode,
  };
}

import { useState, useCallback, useRef } from "react";
import { sendChatMessage, clearChatSession } from "../services/api";
import { v4 as uuidv4 } from "uuid";

export function useChat() {
  const sessionId = useRef(uuidv4());
  const [messages, setMessages] = useState([
    {
      id: "welcome",
      role: "assistant",
      content: "Hi! I can help you analyze the **Order to Cash** process.",
      timestamp: new Date(),
    },
  ]);
  const [loading, setLoading] = useState(false);
  const [lastQueryResult, setLastQueryResult] = useState(null);

  const sendMessage = useCallback(
    async (text, onHighlight) => {
      if (!text.trim() || loading) return;

      const userMsg = {
        id: uuidv4(),
        role: "user",
        content: text,
        timestamp: new Date(),
      };
      setMessages((prev) => [...prev, userMsg]);
      setLoading(true);

      try {
        const res = await sendChatMessage(text, sessionId.current);
        const data = res.data;

        const assistantMsg = {
          id: uuidv4(),
          role: "assistant",
          content: data.answer,
          cypher: data.cypher,
          results: data.results,
          timestamp: new Date(),
        };
        setMessages((prev) => [...prev, assistantMsg]);
        setLastQueryResult(data);

        if (onHighlight && data.highlights?.length > 0) {
          onHighlight(data.highlights);
        }
      } catch (e) {
        const errMsg = {
          id: uuidv4(),
          role: "assistant",
          content: "Sorry, I encountered an error. Please try again.",
          error: true,
          timestamp: new Date(),
        };
        setMessages((prev) => [...prev, errMsg]);
      } finally {
        setLoading(false);
      }
    },
    [loading]
  );

  const clearChat = useCallback(async () => {
    await clearChatSession(sessionId.current);
    sessionId.current = uuidv4();
    setMessages([
      {
        id: "welcome",
        role: "assistant",
        content: "Hi! I can help you analyze the **Order to Cash** process.",
        timestamp: new Date(),
      },
    ]);
    setLastQueryResult(null);
  }, []);

  return {
    messages,
    loading,
    lastQueryResult,
    sendMessage,
    clearChat,
    sessionId: sessionId.current,
  };
}

    } catch {}
  }, []);

  useEffect(() => {
    loadFolders();
    pollStatus();
  }, [loadFolders, pollStatus]);

  const handleIngest = async () => {
    try {
      await startIngestion();
      setIngestionRunning(true);
      setTimeout(pollStatus, 1500);
    } catch (e) {
      alert(e.response?.data?.detail || "Failed to start ingestion");
    }
  };

  const filtered = folders.filter((f) =>
    f.name.toLowerCase().includes(search.toLowerCase())
  );

  const totalSize = folders.reduce(
    (acc, f) => acc + (f.files || []).reduce((s, file) => s + (file.size || 0), 0),
    0
  );

  return (
    <div className="data-browser">
      <div className="db-header">
        <div className="db-breadcrumb">
          Storage / <span>sap-o2c-data</span>
        </div>
        <div className="db-title">
          sap-o2c-data
          <span className="db-count-badge">{folders.length} items</span>
        </div>
        <div className="db-toolbar">
          <input
            className="db-search"
            placeholder="Search folders..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
          />
          <button
            className={`db-ingest-btn ${ingestionRunning ? "running" : ""}`}
            onClick={handleIngest}
            disabled={ingestionRunning}
          >
            {ingestionRunning ? "⚙ Ingesting..." : "⚡ Ingest to Graph"}
          </button>
          <button
            className="db-ingest-btn"
            style={{ background: "#6366f1" }}
            onClick={onNavigateToGraph}
          >
            🕸 Open Graph
          </button>
        </div>
      </div>

      {/* Ingestion Status Bar */}
      {ingestionStatus && ingestionStatus !== "idle" && (
        <div className={`ingestion-status-bar ${ingestionStatus}`}>
          <span
            className={`status-dot ${ingestionStatus}`}
          />
          {ingestionStatus === "running" && "Ingestion pipeline is running..."}
          {ingestionStatus === "completed" && `✓ Ingestion completed successfully — ${ingestionSteps.length} steps`}
          {ingestionStatus === "error" && "✗ Ingestion encountered errors"}
        </div>
      )}

      {ingestionSteps.length > 0 && (
        <div className="progress-steps">
          {ingestionSteps.map((step, i) => (
            <span
              key={i}
              className={`step-badge ${step.status}`}
              title={step.error || ""}
            >
              {step.status === "success" ? "✓" : "✗"} {step.step}
            </span>
          ))}
        </div>
      )}

      <div className="db-table-wrapper">
        <table className="db-table">
          <thead>
            <tr>
              <th style={{ width: 40 }}></th>
              <th>Name</th>
              <th>Files</th>
              <th>Last Modified</th>
              <th>Total Size</th>
            </tr>
          </thead>
          <tbody>
            {filtered.map((folder) => {
              const lastModified = folder.files?.reduce(
                (max, f) => Math.max(max, f.modified || 0),
                0
              );
              const size = folder.files?.reduce(
                (acc, f) => acc + (f.size || 0),
                0
              );
              return (
                <tr key={folder.name}>
                  <td>
                    <span className="folder-icon">📁</span>

                  <td>{lastModified ? formatDate(lastModified) : "-"}</td>
                  <td>{size ? formatBytes(size) : "-"}</td>
                </tr>
              );
            })}
          </tbody>
        </table>
      </div>
    </div>
  );
}

  "SalesOrder", "BillingDocument", "OutboundDelivery",

const MAX_VISIBLE = 14;

export default function NodeTooltip({ node, x, y, connections = 0 }) {
  if (!node) return null;

  const props = node.properties || {};
  const entries = Object.entries(props).filter(
    ([, v]) => v !== null && v !== undefined && v !== ""
  );

  // Sort by priority
  const sorted = [
    ...entries.filter(([k]) => PRIORITY_KEYS.includes(k)),
    ...entries.filter(([k]) => !PRIORITY_KEYS.includes(k)),
  ];

  const visible = sorted.slice(0, MAX_VISIBLE);
  const hidden = sorted.length - MAX_VISIBLE;

  // Position tooltip so it doesn't go off screen
  const style = {
    left: Math.min(x + 16, window.innerWidth - 380),
    top: Math.min(y - 10, window.innerHeight - 400),
  };

  return (
    <div className="node-tooltip" style={style}>
      <div className="node-tooltip-title">{node.label}</div>
      {visible.map(([k, v]) => (
        <div className="node-tooltip-row" key={k}>
          <span className="node-tooltip-key">{k}:</span>
          <span className="node-tooltip-val">{String(v)}</span>
        </div>
      ))}
      {hidden > 0 && (
        <div className="node-tooltip-hidden">
          Additional {hidden} fields hidden for readability
        </div>
      )}
      <div className="node-tooltip-connections">
        Connections: {connections}
      </div>
    </div>
  );
}

function renderMarkdown(text) {
  if (!text) return "";
  return text
    .replace(/\*\*(.*?)\*\*/g, "<strong>\$1</strong>")
    .replace(/\*(.*?)\*/g, "<em>\$1</em>")
    .replace(/`(.*?)`/g, '<code style="background:#eee;padding:1px 4px;border-radius:3px;font-size:11px">\$1</code>')
    .replace(/\n/g, "<br/>");
}

  useEffect,
  useRef,
  useState,
  useCallback,
  forwardRef,
  useImperativeHandle,

import * as d3 from "d3";
import NodeTooltip from "../NodeTooltip/NodeTooltip";
import "./GraphViewer.css";

const LABEL_COLORS = {
  SalesOrder:             "#4a90e2",
  SalesOrderItem:         "#63b3ed",
  SalesOrderScheduleLine: "#90cdf4",
  OutboundDelivery:       "#48bb78",
  OutboundDeliveryItem:   "#9ae6b4",
  BillingDocument:        "#f6ad55",
  BillingDocumentItem:    "#fbd38d",

function getColor(label) {

}

function getRadius(label) {
  const large = ["SalesOrder", "BillingDocument", "BusinessPartner", "OutboundDelivery"];
  const medium = ["JournalEntry", "Payment", "Product", "Plant"];
  if (large.includes(label)) return 8;
  if (medium.includes(label)) return 6;
  return 4;

const GraphViewer = forwardRef(function GraphViewer(
  { graphData, loading, highlightedIds = new Set(), onNodeClick },
  ref
) {
  const svgRef        = useRef(null);
  const containerRef  = useRef(null);
  const simulationRef = useRef(null);
  const zoomRef       = useRef(null);
  const gRef          = useRef(null);

  const [tooltip,      setTooltip]      = useState(null);
  const [showOverlay,  setShowOverlay]  = useState(true);
  const [activeFilters,setActiveFilters]= useState(new Set(NODE_TYPES));
  const [dimensions,   setDimensions]   = useState({ w: 900, h: 600 });

  /* ── expose resetZoom to parent ─────────────────────────────────────────── */
  useImperativeHandle(ref, () => ({
    resetZoom: () => {
      if (svgRef.current && zoomRef.current) {
        d3.select(svgRef.current)
          .transition()
          .duration(600)
          .call(zoomRef.current.transform, d3.zoomIdentity);
      }
    },
  }));

  /* ── resize observer ────────────────────────────────────────────────────── */
  useEffect(() => {
    const el = containerRef.current;
    if (!el) return;
    const ro = new ResizeObserver((entries) => {
      for (const e of entries) {
        const { width, height } = e.contentRect;
        if (width > 0 && height > 0) setDimensions({ w: width, h: height });
      }
    });
    ro.observe(el);
    setDimensions({ w: el.clientWidth, h: el.clientHeight });
    return () => ro.disconnect();
  }, []);

  /* ── main D3 render ─────────────────────────────────────────────────────── */
  const buildGraph = useCallback(() => {
    if (!svgRef.current) return;
    if (!graphData?.nodes?.length) return;

    const { w, h } = dimensions;

    /* filter nodes by active type chips */
    const visibleNodes = graphData.nodes.filter((n) => activeFilters.has(n.label));
    const visibleIds   = new Set(visibleNodes.map((n) => n.id));
    const visibleLinks = (graphData.edges || []).filter(
      (e) =>
        visibleIds.has(String(e.source)) && visibleIds.has(String(e.target))
    );

    /* clone so D3 can mutate x/y */
    const nodes = visibleNodes.map((n) => ({ ...n }));
    const links = visibleLinks.map((e) => ({
      source: String(e.source),
      target: String(e.target),
      type:   e.type,
    }));

    /* ── clear ── */
    d3.select(svgRef.current).selectAll("*").remove();

    const svg = d3
      .select(svgRef.current)
      .attr("width",  w)
      .attr("height", h)
      .style("cursor", "grab");

    /* ── defs: arrowhead ── */
    const defs = svg.append("defs");
    defs
      .append("marker")
      .attr("id",          "arrow")
      .attr("viewBox",     "0 -4 8 8")
      .attr("refX",        16)
      .attr("refY",        0)
      .attr("markerWidth", 4)
      .attr("markerHeight",4)
      .attr("orient",      "auto")
      .append("path")
      .attr("d",    "M0,-4L8,0L0,4")
      .attr("fill", "#b0c8e8")
      .attr("opacity", 0.6);

    /* ── zoom ── */
    const g = svg.append("g");
    gRef.current = g;

    const zoom = d3
      .zoom()
      .scaleExtent([0.03, 10])
      .on("zoom", (event) => g.attr("transform", event.transform));
    zoomRef.current = zoom;
    svg.call(zoom);

    /* ── links ── */
    const linkSel = g
      .append("g")
      .selectAll("line")
      .data(links)
      .join("line")
      .attr("stroke",       "#b0c8e8")
      .attr("stroke-opacity", 0.5)
      .attr("stroke-width", 1)
      .attr("marker-end",   "url(#arrow)");

    /* ── nodes ── */
    const nodeSel = g
      .append("g")
      .selectAll("g")
      .data(nodes, (d) => d.id)
      .join("g")
      .style("cursor", "pointer")
      .call(
        d3
          .drag()
          .on("start", (event, d) => {
            if (!event.active)
              simulationRef.current?.alphaTarget(0.3).restart();
            d.fx = d.x;
            d.fy = d.y;
          })
          .on("drag", (event, d) => {
            d.fx = event.x;
            d.fy = event.y;
          })
          .on("end", (event, d) => {
            if (!event.active)
              simulationRef.current?.alphaTarget(0);
            d.fx = null;
            d.fy = null;
          })
      );

    /* circle */
    nodeSel
      .append("circle")
      .attr("r",           (d) => getRadius(d.label))
      .attr("fill",        (d) => getColor(d.label))
      .attr("stroke",      (d) =>
        highlightedIds.has(d.id) ? "#f59e0b" : "rgba(255,255,255,0.4)"
      )
      .attr("stroke-width",(d) => (highlightedIds.has(d.id) ? 3 : 1));

    /* label – only for highlighted or large nodes */
    nodeSel
      .append("text")
      .attr("dy",        "0.35em")
      .attr("x",         (d) => getRadius(d.label) + 3)
      .attr("font-size", "9px")
      .attr("fill",      "#555")
      .attr("pointer-events", "none")
      .text((d) =>
        highlightedIds.has(d.id)
          ? d.display_id?.slice(0, 12) || d.label
          : ""
      );

    /* hover tooltip */
    nodeSel
      .on("mouseover", (event, d) => {
        const [px, py] = d3.pointer(event, containerRef.current);
        // Count connections
        const connCount = links.filter(
          (l) =>
            String(l.source.id || l.source) === d.id ||
            String(l.target.id || l.target) === d.id
        ).length;
        setTooltip({ node: d, x: px, y: py, connections: connCount });
      })
      .on("mousemove", (event) => {
        const [px, py] = d3.pointer(event, containerRef.current);
        setTooltip((prev) => prev ? { ...prev, x: px, y: py } : prev);
      })
      .on("mouseout", () => setTooltip(null))
      .on("click", (event, d) => {
        event.stopPropagation();
        onNodeClick && onNodeClick(d);
      });

    /* ── simulation ── */
    if (simulationRef.current) simulationRef.current.stop();

    const sim = d3
      .forceSimulation(nodes)
      .force(
        "link",
        d3
          .forceLink(links)
          .id((d) => d.id)
          .distance(60)
          .strength(0.4)
      )
      .force("charge", d3.forceManyBody().strength(-120).distanceMax(300))
      .force("center", d3.forceCenter(w / 2, h / 2))
      .force("collision", d3.forceCollide().radius((d) => getRadius(d.label) + 3))
      .alphaDecay(0.025);

    simulationRef.current = sim;

    sim.on("tick", () => {
      linkSel
        .attr("x1", (d) => d.source.x)
        .attr("y1", (d) => d.source.y)
        .attr("x2", (d) => d.target.x)
        .attr("y2", (d) => d.target.y);

      nodeSel.attr("transform", (d) => `translate(${d.x},${d.y})`);
    });

    /* initial zoom to fit after simulation cools */
    sim.on("end", () => {
      const padding = 40;
      const xs = nodes.map((n) => n.x);
      const ys = nodes.map((n) => n.y);
      const x0 = Math.min(...xs) - padding;
      const y0 = Math.min(...ys) - padding;
      const x1 = Math.max(...xs) + padding;
      const y1 = Math.max(...ys) + padding;
      const bw = x1 - x0 || 1;
      const bh = y1 - y0 || 1;
      const scale = Math.min(0.9, Math.min(w / bw, h / bh));
      const tx = (w - scale * (x0 + x1)) / 2;
      const ty = (h - scale * (y0 + y1)) / 2;
      svg
        .transition()
        .duration(800)
        .call(
          zoom.transform,
          d3.zoomIdentity.translate(tx, ty).scale(scale)
        );
    });

    return () => sim.stop();
  }, [graphData, dimensions, activeFilters, highlightedIds, onNodeClick]);

  /* rebuild whenever deps change */
  useEffect(() => {
    buildGraph();
  }, [buildGraph]);

  /* update circle colours when highlights change without full rebuild */
  useEffect(() => {
    if (!gRef.current) return;
    gRef.current
      .selectAll(".graph-node circle, g circle")
      .attr("stroke",       (d) =>
        d && highlightedIds.has(d.id) ? "#f59e0b" : "rgba(255,255,255,0.4)"
      )
      .attr("stroke-width", (d) =>
        d && highlightedIds.has(d.id) ? 3 : 1
      )
      .attr("r",            (d) =>
        d ? getRadius(d.label) + (highlightedIds.has(d.id) ? 3 : 0) : 4
      );

    gRef.current
      .selectAll("g text")
      .text((d) =>
        d && highlightedIds.has(d.id)
          ? d.display_id?.slice(0, 12) || d.label
          : ""
      );
  }, [highlightedIds]);

  /* toggle filter chip */
  const toggleFilter = useCallback((label) => {
    setActiveFilters((prev) => {
      const next = new Set(prev);
      next.has(label) ? next.delete(label) : next.add(label);
      return next;
    });
  }, []);

  /* label counts */
  const labelCounts = {};
  (graphData?.nodes || []).forEach((n) => {
    labelCounts[n.label] = (labelCounts[n.label] || 0) + 1;
  });
  const presentLabels = NODE_TYPES.filter((l) => labelCounts[l]);

  return (
    <div className="graph-viewer" ref={containerRef}>
      {/* Toolbar */}
      <div className="graph-toolbar">
        <button
          className="graph-btn"
          onClick={() => {
            if (svgRef.current && zoomRef.current) {
              d3.select(svgRef.current)
                .transition()
                .duration(500)
                .call(zoomRef.current.transform, d3.zoomIdentity);
            }
          }}
        >
          ⤢ Minimize
        </button>
        <button
          className={`graph-btn ${showOverlay ? "active" : ""}`}
          onClick={() => setShowOverlay((v) => !v)}
        >
          🔲 {showOverlay ? "Hide" : "Show"} Granular Overlay
        </button>
      </div>

      {/* Node-type filter chips */}
      {showOverlay && (
        <div className="node-type-filters">
          {presentLabels.map((label) => (
            <span
              key={label}
              className={`filter-chip ${activeFilters.has(label) ? "active" : ""}`}
              style={{ background: getColor(label), color: "#333" }}
              onClick={() => toggleFilter(label)}
              title={`${labelCounts[label] || 0} nodes`}
            >
              {label.replace(/([A-Z])/g, " \$1").trim()}
            </span>
          ))}
        </div>
      )}

      {/* SVG canvas */}
      <svg ref={svgRef} className="graph-canvas" />

      {/* Loading overlay */}
      {loading && (
        <div className="graph-loading">
          <div className="graph-spinner" />
          <span className="graph-loading-text">Loading graph data...</span>
        </div>
      )}

      {/* Tooltip */}
      {tooltip && (
        <NodeTooltip
          node={tooltip.node}
          x={tooltip.x}
          y={tooltip.y}
          connections={tooltip.connections}
        />
      )}

      {/* Legend */}
      {showOverlay && presentLabels.length > 0 && (
        <div className="graph-legend">
          <div className="legend-title">Node Types</div>
          {presentLabels.slice(0, 8).map((label) => (
            <div className="legend-item" key={label}>
              <div
                className="legend-dot"
                style={{ background: getColor(label) }}
              />
              {label.replace(/([A-Z])/g, " \$1").trim()}
            </div>
          ))}
          {presentLabels.length > 8 && (
            <div className="legend-item" style={{ color: "#aaa" }}>
              + {presentLabels.length - 8} more
            </div>
          )}
        </div>
      )}

      {/* Stats bar */}
      <div className="graph-stats-bar">
        <strong>{(graphData?.nodes || []).length}</strong> nodes &nbsp;|&nbsp;
        <strong>{(graphData?.edges || []).length}</strong> edges
        {highlightedIds.size > 0 && (
          <> &nbsp;|&nbsp; <strong style={{ color: "#f59e0b" }}>{highlightedIds.size}</strong> highlighted</>
        )}
      </div>
    </div>
  );
});

                onClick={() => loadGraph(400)}
              >
                ↻ Reload
              </button>
            </>
          )}
          <button
            className="topbar-action-btn"
            onClick={() =>
              setActiveView((v) => (v === "browser" ? "graph" : "browser"))
            }
          >
            {activeView === "browser" ? "🕸 Graph View" : "🗂 Data Browser"}
          </button>
        </div>
      </div>

      {/* ── Main layout ── */}
      <div className="app-layout">
        <Sidebar
          activeView={activeView}
          onViewChange={setActiveView}
          collapsed={sidebarCollapsed}
          stats={stats}
        />

        <div className="main-content">
          {activeView === "browser" ? (
            <DataBrowser onNavigateToGraph={() => setActiveView("graph")} />
          ) : (
            <div className="graph-layout">
              <GraphViewer
                ref={graphRef}
                graphData={graphData}
                loading={graphLoading}
                highlightedIds={highlightedIds}
                onNodeClick={handleNodeClick}
              />
              <ChatPanel
                messages={messages}
                loading={chatLoading}
                onSend={handleSend}
                onClear={clearChat}
              />
            </div>
          )}
        </div>
      </div>

      {/* ── Toast ── */}
      {toast && (
        <Toast message={toast} onDone={() => setToast(null)} />
      )}
    </>
  );
}

## 1. Project Overview

The SAP Order-to-Cash (O2C) process spans many documents and systems:
a customer places a sales order → goods are picked and delivered →
an invoice (billing document) is raised → accounting entries are posted →
payment is received and matched.

This platform:

- **Ingests** all 19 SAP O2C CSV data folders into Neo4j, modelling every
  document as a node and every business relationship as a graph edge.
- **Visualises** the resulting graph as an interactive, filterable,
  force-directed diagram built with D3.js.
- **Enables plain-English queries** through a chat panel powered by GPT-4o,
  which translates questions into Cypher queries, executes them, and returns
  business-language answers.
- **Highlights** relevant nodes in the graph in real time as the agent answers
  each question.
- **Guards** against misuse through a multi-layer guardrail system that blocks
  off-topic prompts, dangerous Cypher, and prompt-injection attempts.

## 5. Backend — File-by-File

### 5.1 `main.py`

**What it does**

The application entry point. Creates the FastAPI app, registers all three
routers, configures CORS and GZip middleware, and manages the Neo4j driver
lifecycle through a lifespan context manager (connects on startup, disconnects
on shutdown). Also exposes a `/api/folders` endpoint that lists the data
directory contents so the front-end Data Browser can display real file
metadata.

**Key details**

- `lifespan` async context manager ensures clean DB connection teardown even if
  the server is killed with SIGTERM.
- `GZipMiddleware` compresses API responses larger than 1 KB — important when
  the graph overview returns hundreds of nodes.
- `CORSMiddleware` reads allowed origins from settings so the same binary works
  in local dev (`localhost:3000`) and production.

---

### 5.2 `config.py`

**What it does**

Centralises every configurable value using `pydantic-settings`. Values are read
from environment variables or a `.env` file. The `@lru_cache` decorator ensures
the settings object is created only once no matter how many modules import it.

**Settings**

| Variable | Default | Purpose |
|---|---|---|
| `NEO4J_URI` | `bolt://localhost:7687` | Neo4j Bolt connection string |
| `NEO4J_USER` | `neo4j` | Database username |
| `NEO4J_PASSWORD` | `password` | Database password |
| `OPENAI_API_KEY` | _(required)_ | Your OpenAI secret key |
| `OPENAI_MODEL` | `gpt-4o` | Model used for Cypher generation and answers |
| `DATA_DIR` | `./data/sap-o2c-data` | Root folder containing the 19 CSV subfolders |
| `CORS_ORIGINS` | `["http://localhost:3000"]` | Allowed browser origins |

---

### 5.3 `database/neo4j_client.py`

**What it does**

Manages a single shared async Neo4j driver instance (singleton pattern).
Provides two helper functions:

- `run_query(cypher, params)` — executes a read query and returns a list of
  dictionaries.
- `run_write_query(cypher, params)` — executes a write query and returns the
  result summary (counters, etc.).

**Why a singleton?**
  
The Neo4j async driver maintains an internal connection pool. Creating a new
driver per request would exhaust TCP sockets under load. The singleton is
created once at startup and reused for all requests.

---

### 5.4 `database/schema.py`

**What it does**

Two things:

1. **`SCHEMA_INDEXES`** — a list of Cypher `CREATE INDEX` statements run once
   during startup ingestion. Indexes are created on the primary identifier
   property of each node label (e.g. `SalesOrder.SalesOrder`,
   `BillingDocument.BillingDocument`). Without these, every node lookup is a
   full scan.

2. **`SCHEMA_DESCRIPTION`** — a multi-line string injected into every LLM
   prompt. It tells GPT-4o the exact node labels, their key properties, and
   the available relationship types so the model generates correct Cypher
   without hallucinating field names.

### 5.5 `ingestion/data_loader.py`

**What it does**

Provides generic utilities to turn files on disk into clean Python lists of
dictionaries:

- `load_csv_folder(folder_name)` — recursively finds all `.csv` and `.parquet`
  files in a named subfolder of `DATA_DIR`, loads each one with Pandas, and
  concatenates them into a single DataFrame.
- `clean_row(row)` — converts `NaN` values to `None` so they become JSON `null`
  rather than the string `"nan"` when sent to Neo4j.
- `df_to_records(df)` — calls `clean_row` on every row and returns a plain
  Python list of dicts.

**Why Pandas?**

SAP exports are often multiple part-files per entity (Parquet partitions or
CSV chunks). Pandas handles both formats, concats them automatically, and
provides the `NaN` semantics needed for clean graph writes.

---

### 5.6 `ingestion/graph_builder.py`

**What it does**

Orchestrates the full ingestion pipeline. Contains one loader function per
SAP entity (19 in total) plus relationship-wiring functions. Everything is
batched in groups of 500 rows using `UNWIND … MERGE` Cypher so large datasets
don't create thousands of individual round-trips.

**Key design decisions**

| Decision | Reason |
|---|---|
| `MERGE` instead of `CREATE` | Idempotent — re-running ingestion updates existing nodes rather than duplicating them |
| Composite key properties | SAP items need compound keys (`SalesOrder + SalesOrderItem`), so a synthetic `_Key` property is constructed before upserting |
| Two-pass relationship creation | Nodes are all created first; then relationship queries `MATCH` both ends. This avoids ordering problems where side A exists but side B hasn't been loaded yet |
| Batch size 500 | Balances transaction size against memory pressure in Neo4j |
| Background task | The full pipeline runs in a FastAPI `BackgroundTasks` callback so the HTTP response returns immediately with a "started" status |

### 5.7 `llm/prompts.py`

**What it does**

Stores every prompt template as a module-level constant. No prompt strings
appear anywhere else in the codebase — all modifications to LLM behaviour are
made here.

**Templates**

| Constant | Used for |
|---|---|
| `SYSTEM_PROMPT` | Injected as the system message in every OpenAI call. Describes Dodge AI's persona, capabilities, rules, and the full schema |
| `CYPHER_GENERATION_PROMPT` | Asks GPT-4o to translate a user question into a JSON object containing a Cypher query and an explanation |
| `ANSWER_GENERATION_PROMPT` | Given the executed query results, asks GPT-4o to produce a business-language answer |
| `HIGHLIGHT_PROMPT` | (utility) Asks GPT-4o to identify which result nodes should be highlighted in the graph |

---

### 5.8 `llm/guardrails.py`

**What it does**

Provides three independent safety functions that are called at different points
in the request lifecycle.

**Layer 1 — `check_message_safety(message)`**

Called before any LLM API call. Checks the raw user message against:

- **Blocked regex patterns** — passwords, API keys, credit card numbers,
  jailbreak phrases, weather, jokes, sports and other clearly off-topic
  domains.
- **Topic relevance check** — for messages longer than five words, at least
  one word must match a list of ~40 O2C-related keywords (order, billing,
  invoice, delivery, plant, journal, etc.). Messages with no relevant terms
  are politely redirected.

Returns `(is_safe: bool, reason: str)`.

**Layer 2 — `validate_cypher_query(cypher)`**

Called after the LLM generates a Cypher query, before execution. Blocks:

- `DELETE`, `DETACH DELETE`, `DROP`
- `CREATE USER`, `CREATE DATABASE`
- `SET PASSWORD`, `CALL dbms.*`
- Any query that doesn't contain at least one of `MATCH`, `RETURN`, `WITH`,
  `CALL`, `UNWIND` (ensures it is a read or safe aggregation query).

**Layer 3 — `sanitize_cypher_limit(cypher, max_limit=500)`**

Appends `LIMIT 500` to any query that doesn't already have a `LIMIT` clause.
Prevents a `MATCH (n) RETURN n` from returning the entire database.

### 5.9 `llm/agent.py`

**What it does**

Contains the `GraphAgent` class — the core intelligence of the platform.

**Full request lifecycle**
User message
│
▼
check_message_safety() ← guardrail layer 1
│ safe
▼
_generate_cypher() ← GPT-4o call 1
└─ CYPHER_GENERATION_PROMPT
└─ returns {cypher, explanation}
│
▼
validate_cypher_query() ← guardrail layer 2
sanitize_cypher_limit() ← guardrail layer 3
│
▼
run_query(cypher) ← Neo4j execution
│ on error → auto-fix retry (GPT-4o call 1b)
▼
_generate_answer() ← GPT-4o call 2
└─ ANSWER_GENERATION_PROMPT
└─ last 3 exchanges from history injected for context
│
▼
_extract_highlight_ids() ← maps result props → graph node IDs
│
▼
return {answer, cypher, results, highlights}

text

**Session management**

Each browser tab gets a UUID session ID. `get_agent(session_id)` returns the
same `GraphAgent` instance for that session, preserving `conversation_history`
(last 6 messages = 3 exchanges) across turns. Sessions are stored in a
module-level dict `_agents`.

**Retry logic**

`_call_llm` is decorated with `@retry(stop_after_attempt(3), wait_exponential)`
from Tenacity. If the OpenAI API returns a rate-limit or transient error, it
waits 1→2→4 seconds and retries automatically.

### 6.9 `src/components/GraphViewer`

**GraphViewer.jsx**

The most complex component. Renders an interactive D3 force-directed graph
inside an SVG element that fills its container.

**How it works step by step**

1. **Resize observer** — A `ResizeObserver` on the container div tracks width
   and height. When dimensions change, the D3 simulation is rebuilt at the
   new size.

2. **Data filtering** — Before building the simulation, nodes are filtered by
   the active filter chips (node type toggles). Edges whose endpoints are no
   longer visible are also removed.

3. **SVG construction** — A `<g>` group is appended and a D3 zoom behaviour is
   attached to the SVG. All graph elements go inside the group so zoom/pan
   affects them uniformly.

4. **Arrowhead marker** — A `<defs>/<marker>` element adds a small arrowhead
   at the target end of every edge line.

5. **Force simulation** — Five forces:
   - `forceLink` — pulls connected nodes toward each other (distance 60px).
   - `forceManyBody` — repels all nodes (strength −120, max distance 300px).
   - `forceCenter` — keeps the cluster centred in the viewport.
   - `forceCollide` — prevents node overlap.
   - `alphaDecay 0.025` — slower cooling for a more spread-out layout.

6. **Tick handler** — On every simulation tick, `line` x1/y1/x2/y2 and `g`
   `transform` attributes are updated to move nodes and edges.

7. **Drag behaviour** — `d3.drag` allows individual nodes to be pinned and
   repositioned. Dragging restarts the simulation with `alphaTarget(0.3)`.

8. **Tooltips** — `mouseover`/`mousemove`/`mouseout` events on node groups
   update `tooltip` React state, which renders `<NodeTooltip>` at the pointer
   position.

9. **Highlights** — A `useEffect` watching `highlightedIds` updates circle
   stroke colour (amber), stroke width, and radius in place without rebuilding
   the simulation. Text labels appear only for highlighted nodes.

10. **Auto-fit zoom** — After the simulation ends (`sim.on("end", ...)`) the
    bounding box of all nodes is calculated and a zoom transform is computed
    to fit the graph into the viewport with 40px padding.

11. **`useImperativeHandle`** — Exposes `resetZoom()` to the parent via a ref
    so the "⤢ Fit Graph" topbar button can trigger a zoom reset.

**GraphViewer.css**

SVG cursor is `grab`. Node circles transition radius on hover. Legend and stats
bar are absolutely positioned over the canvas with a frosted-glass effect
(`backdrop-filter: blur`).

## 8. LLM Integration

### 8.1 How Natural Language Becomes a Cypher Query
Step 1 — User types:
"91150187 - Find the journal entry number linked to this?"

Step 2 — check_message_safety() passes (contains "journal", "entry")

Step 3 — GPT-4o call with CYPHER_GENERATION_PROMPT:
Input: question + full schema description
Output: {
"cypher": "MATCH (j:JournalEntry)-[:RECORDS]->(b:BillingDocument
{BillingDocument: '91150187'}) RETURN j.AccountingDocument",
"explanation": "Finds journal entries that record billing document 91150187"
}

Step 4 — validate_cypher_query() passes (read-only MATCH)
sanitize_cypher_limit() adds LIMIT 500

Step 5 — run_query() executes against Neo4j
Returns: [{"j.AccountingDocument": "9400635958"}]

Step 6 — GPT-4o call with ANSWER_GENERATION_PROMPT:
Input: original question + cypher + results (first 20 rows)
Output: "The journal entry number linked to billing document
91150187 is 9400635958."

Step 7 — _extract_highlight_ids() scans results for node properties,
matches them against graphData.nodes in the browser,
sets highlighted node IDs (amber glow in D3)

Step 8 — Response returned:
{answer, cypher, results, highlights}

### 8.3 Session Memory

Each browser session gets a UUID. The corresponding `GraphAgent` keeps the
last 6 messages (3 user + 3 assistant) in `conversation_history`. These are
injected into every LLM call so the agent can answer follow-up questions like
"What about the delivery for that order?" without the user needing to repeat
context.

Node colours are deterministic per label type (defined in LABEL_COLORS).
Node sizes are 8px for primary documents, 6px for secondary, 4px for
detail nodes. Edges are semi-transparent blue with arrowheads.

16. Evaluation Criteria Map
Criterion	Where it is implemented
Code quality and architecture	Separation of concerns: api/, llm/, ingestion/, database/ are independent modules. Pydantic models on all boundaries. Async throughout. Single-responsibility functions.
Graph modelling	database/schema.py defines 19 node labels and 18 relationship types that faithfully represent real SAP O2C document flows. Compound keys for child entities. Indexes on all primary identifiers.
Database / storage choice	Neo4j chosen because O2C is fundamentally a graph: documents reference other documents across process steps. Graph traversal (e.g. "trace this billing doc back to its sales order") is a single Cypher path query vs. 5 SQL joins.
LLM integration and prompting	Three-prompt pipeline: schema-aware Cypher generation → safe execution → NL answer synthesis. Auto-fix retry on query error. Conversation history for follow-ups. Schema description injected verbatim so GPT-4o never guesses field names.
Guardrails	Three independent layers: message safety (regex + keyword), Cypher validation (dangerous operation blocking), automatic LIMIT injection. Off-topic messages are caught before any LLM call is made.
17. Troubleshooting
Neo4j connection refused
bash
# Check Neo4j is running
docker ps | grep neo4j
# Check bolt port
nc -zv localhost 7687
# View Neo4j logs
docker logs neo4j
"Ingestion already running"
The background task is still active. Poll /api/ingestion/status until
status is completed or error, then you can restart.

OpenAI API errors
Verify OPENAI_API_KEY in backend/.env.
Check you have GPT-4o access on your account.
The agent automatically retries 3 times with exponential back-off for
transient errors.
Graph shows no nodes
Check ingestion completed: GET /api/ingestion/status
Verify data files exist in data/sap-o2c-data/
Check Neo4j browser at http://localhost:7474:
MATCH (n) RETURN count(n) should return > 0
Frontend shows CORS error
Ensure CORS_ORIGINS in backend/.env includes your frontend URL exactly:

env
CORS_ORIGINS=["http://localhost:3000"]
D3 graph is empty but nodes loaded
Check browser console for errors. Most common cause: node id values are
returned as numbers from Neo4j but D3 forceLink expects string IDs. The
fetchGraphOverview response maps all IDs to strings before returning.

18. Extending the System
Add a new SAP entity
Add a loader function in ingestion/graph_builder.py.
Add the label and key to database/schema.py (SCHEMA_INDEXES and
SCHEMA_DESCRIPTION).
Add the colour to LABEL_COLORS in GraphViewer.jsx.
Call the loader in run_full_ingestion().
Change the LLM model
Set OPENAI_MODEL=gpt-4-turbo (or any chat completion model) in .env.
The agent passes it directly to the OpenAI client.

Add streaming chat responses
Replace the sendChatMessage Axios POST with a fetch call consuming
text/event-stream. On the backend, replace return ChatResponse(...) with
a StreamingResponse using openai.chat.completions.create(stream=True).

Connect to a different graph database
Replace database/neo4j_client.py with a client for ArangoDB, Amazon Neptune,
or any database that speaks Cypher or Gremlin. Update schema.py with the
new query dialect.

Export graph as image
Add a button in GraphViewer.jsx that calls:

javascript
const svgEl = svgRef.current;
const serializer = new XMLSerializer();
const svgStr = serializer.serializeToString(svgEl);
const blob = new Blob([svgStr], {type: "image/svg+xml"});
const url = URL.createObjectURL(blob);
const a = document.createElement("a");
a.href = url; a.download = "graph.svg"; a.click();
